CREATE TABLE `mt_user` (
  `USERNAME` varchar(10) NOT NULL,
  `PASSWORD` varchar(100) DEFAULT NULL,
  `USER_FIRST_NAME` varchar(40) NOT NULL,
  `USER_LAST_NAME` varchar(40) NOT NULL,
  `ABOUT_USER` varchar(50) DEFAULT NULL,
  `EMAIL` varchar(30) NOT NULL,
  `DOB` date NOT NULL,
  `USER_NODE` varchar(40) NOT NULL,
  `CONTENT_LOCATION` varchar(200) DEFAULT NULL,
  `CONTENT_FORMAT` varchar(20) NOT NULL,
  PRIMARY KEY (`USERNAME`)
);

CREATE TABLE `mt_tweet` (
  `TWEET_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `USERNAME` varchar(10) NOT NULL,
  `TWEET` varchar(100) NOT NULL,
  PRIMARY KEY (`TWEET_ID`),
  KEY `USERNAME` (`USERNAME`),
  CONSTRAINT `mt_tweet_ibfk_1` FOREIGN KEY (`USERNAME`) REFERENCES `mt_user` (`USERNAME`)
);

CREATE TABLE `mt_following` (
  `FOLLOWING_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `USERNAME` varchar(10) NOT NULL,
  `FOLLOWING_USERNAME` varchar(10) NOT NULL,
  PRIMARY KEY (`FOLLOWING_ID`),
  KEY `USERNAME` (`USERNAME`),
  KEY `FOLLOWING_USERNAME` (`FOLLOWING_USERNAME`),
  CONSTRAINT `mt_following_ibfk_1` FOREIGN KEY (`USERNAME`) REFERENCES `mt_user` (`USERNAME`),
  CONSTRAINT `mt_following_ibfk_2` FOREIGN KEY (`FOLLOWING_USERNAME`) REFERENCES `mt_user` (`USERNAME`)
);

CREATE TABLE `mt_followers` (
  `FOLLOWER_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `USERNAME` varchar(10) NOT NULL,
  `FOLLOWER_USERNAME` varchar(10) NOT NULL,
  PRIMARY KEY (`FOLLOWER_ID`),
  KEY `USERNAME` (`USERNAME`),
  KEY `FOLLOWER_USERNAME` (`FOLLOWER_USERNAME`),
  CONSTRAINT `mt_followers_ibfk_1` FOREIGN KEY (`USERNAME`) REFERENCES `mt_user` (`USERNAME`),
  CONSTRAINT `mt_followers_ibfk_2` FOREIGN KEY (`FOLLOWER_USERNAME`) REFERENCES `mt_user` (`USERNAME`)
);

CREATE TABLE `mt_tweets` (
  `TWEET_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `TWEET_SEQ_NO` int(11) NOT NULL,
  `USERNAME` varchar(10) NOT NULL,
  `TWEET_DATA` varchar(300) NOT NULL,
  `TWEET_INDEX` int(10) NOT NULL,
  `TWEET_TIME` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`TWEET_ID`),
  KEY `USERNAME` (`USERNAME`),
  CONSTRAINT `mt_tweets_ibfk_1` FOREIGN KEY (`USERNAME`) REFERENCES `mt_user` (`USERNAME`)
);
